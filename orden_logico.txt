  -H 'Content-Type: application/json' \
  -H 'fiware-service: pamplona' \
  -H 'fiware-servicepath: /iot' \
  -d '{
 "services": [
   {
     "apikey":      "11111111111",

     "resource":    "/iot/json"

 ]
}'
