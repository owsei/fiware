  -H 'Content-Type: application/json' 
  -H "Fiware-Service: pamplona" 
  -H "Fiware-ServicePath: /" \
-d '{
  "description": "Notify Cygnus of changes in IoT Agent",
  "subject": {
    "entities": [
      {
        "idPattern": ".*"
      }
    ],
    "condition": {
      "attrs": ["temperature"]
    }
  },
  "notification": {
    "http": {
      "url": "http://cygnus:5080/v2/notify"
    },
    "attrsFormat": "values"
  },
  "throttling": 0
}'
